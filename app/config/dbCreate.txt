DROP DATABASE if exists SAT;
CREATE DATABASE SAT;
USE SAT;

-- CREATE TABLES
CREATE TABLE Users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    password VARCHAR(255),
    email VARCHAR(255) UNIQUE,
    name VARCHAR(255),
    isAdmin BOOLEAN,
    createdAt DATETIME,
    updatedAt DATETIME,
    deletedAt DATETIME NULL
);

CREATE TABLE Icons (
    id INT AUTO_INCREMENT PRIMARY KEY,
    src VARCHAR(255) UNIQUE,
    name VARCHAR(255) UNIQUE,
    alt TEXT,
    createdAt DATETIME,
    updatedAt DATETIME,
    deletedAt DATETIME NULL
);

CREATE TABLE Templates (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255) UNIQUE,
    defaultValue DECIMAL(10, 2),
    depreciationRate DECIMAL(10, 2),
    defaultUsefulLife INT,
    icon INT,
    singleResident BOOLEAN,
    createdAt DATETIME,
    updatedAt DATETIME,
    deletedAt DATETIME NULL,
    FOREIGN KEY (icon) REFERENCES Icons(id)
);

CREATE TABLE Facilities (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255) UNIQUE,
    managerId INT,
    createdAt DATETIME,
    updatedAt DATETIME,
    deletedAt DATETIME NULL,
    FOREIGN KEY (managerId) REFERENCES Users(id)
);

CREATE TABLE Units (
    id INT AUTO_INCREMENT PRIMARY KEY,
    facilityId INT,
    name VARCHAR(255) UNIQUE,
    type VARCHAR(255),
    createdAt DATETIME,
    updatedAt DATETIME,
    deletedAt DATETIME NULL,
    FOREIGN KEY (facilityId) REFERENCES Facilities(id)
);

CREATE TABLE FacilityAuths (
    id INT AUTO_INCREMENT PRIMARY KEY,
    userId INT,
    facilityId INT,
    authorizedBy INT,
    createdAt DATETIME,
    deletedAt DATETIME NULL,
    FOREIGN KEY (userId) REFERENCES Users(id),
    FOREIGN KEY (facilityId) REFERENCES Facilities(id),
    FOREIGN KEY (authorizedBy) REFERENCES Users(id)
);

CREATE TABLE Items (
    id INT AUTO_INCREMENT PRIMARY KEY,
    unitId INT,
    templateId INT,
    name VARCHAR(255),
    donated BOOLEAN,
    vendor VARCHAR(255),
    invoice VARCHAR(255),
    initialValue DECIMAL(10, 2),
    usefulLife DATETIME,
    status ENUM('ok', 'inspect', 'discard'),
    addedBy INT,
    createdAt DATETIME,
    updatedAt DATETIME,
    deletedAt DATETIME NULL,
    FOREIGN KEY (unitId) REFERENCES Units(id),
    FOREIGN KEY (templateId) REFERENCES Templates(id),
    FOREIGN KEY (addedBy) REFERENCES Users(id)
);

CREATE TABLE Inspections (
    id INT AUTO_INCREMENT PRIMARY KEY,
    userId INT,
    itemId INT,
    comment TEXT,
    createdAt DATETIME,
    deletedAt DATETIME,
    FOREIGN KEY (userId) REFERENCES Users(id),
    FOREIGN KEY (itemId) REFERENCES Items(id)
);
